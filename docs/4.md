#### 看个例子

```c
/**
 * 1. 输出指针指向的地址
 * 2. 将指针指向的内存解读为 long，并以 16 进制 16 位格式输出
 */
void printPtr(void * ptr) {
	printf("ptr = %p, *ptr = %016lx\n", ptr, *(long *)ptr);
}

int main(int argc, char const * argv[]) {

	int x;
	printPtr(&x);
	printPtr(&argc);

	printPtr(main);
	printPtr(&main);

	printPtr(argv);
	printPtr(&argv);
}

/*
	结论：
	
		1. 一切皆可取地址
		2. C 语言代码中的所有东西都可以看成指针
*/
```
